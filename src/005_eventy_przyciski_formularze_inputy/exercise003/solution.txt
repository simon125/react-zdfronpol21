 * Twoim zadaniem jest stworzenie aplikacji typu kalkulator na podstawie designu z pliku desing.png
 * User jest w stanie wpisaÄ‡ dwie liczby w dwa inputy, moÅ¼e rÃ³wnieÅ¼ okreÅ›liÄ‡ dziaÅ‚anie matematyczne

 * do wpisania liczb uÅ¼yj zwykÅ‚ych inputÃ³w tekstowych natomiast do wyboru
 * dziaÅ‚ania matematycznego moÅ¼esz uÅ¼yÄ‡ selecta z optionami - pamiÄ™taj Å¼eby kaÅ¼dy option oprÃ³cz labelki
 * miaÅ‚ rÃ³wnieÅ¼ okreÅ›lonÄ… wartoÅ›Ä‡ (atrybut value)
 *
 * do powyÅ¼szych rzeczy (liczba1, liczba2, dziaÅ‚anie matematyczne) bÄ™dziesz potrzebowaÄ‡ stanu lub kilku stanÃ³wğŸ˜‰ (useState)
 * wynik dziaÅ‚ania natomiast bÄ™dzie tzw computed value - bÄ™dzie budowana przy kaÅ¼dym rerenderze na nowo
 * na podstawie 3 w/w stanÃ³w pamiÄ™taj user moÅ¼e chieÄ‡ dodaÄ‡, odjÄ…Ä‡, podzieliÄ‡ lub pomnoÅ¼yÄ‡ wpisane liczby.
 *
 * do logiki (obliczenia) i zwrÃ³cenia wyniki obliczenia mozesz uÅ¼yÄ‡ tworzÄ…c dodatkowa funkcje wewnÄ…trz lub na zewnÄ…trz komponentu

 *

export const Calculator = () => {
  const [num1, setNum1] = useState("");
  const [num2, setNum2] = useState("");
  const [operation, setOperation] = useState("");

  const getResult = (n1, n2, op) => {
    if (op === "+") {
      return n1 + n2;
    } else if (op === "-") {
      return n1 - n2;
    } else if (op === "/") {
      return n1 / n2;
    } else if (op === "*") {
      return n1 * n2;
    }

    return "";
  };

  const result = getResult(Number(num1), Number(num2), operation);

  return (
    <article>
      <p>Calculator</p>
      <div style={{ display: "flex", gap: 20 }}>
        <input
          type="text"
          placeholder="Liczba pierwsza..."
          value={num1}
          onChange={(e) => {
            setNum1(e.target.value);
          }}
        />
        <select
          name=""
          id=""
          value={operation}
          onChange={(e) => {
            setOperation(e.target.value);
          }}
        >
          <option value="" hidden>
            Wybierz dziaÅ‚anie
          </option>
          <option value="+">+</option>
          <option value="-">-</option>
          <option value="/">/</option>
          <option value="*">*</option>
        </select>
        <input
          type="text"
          placeholder="Liczba druga..."
          value={num2}
          onChange={(e) => {
            setNum2(e.target.value);
          }}
        />
        <input type="text" placeholder="Wynik" value={result} readOnly />
      </div>
    </article>
  );
};
